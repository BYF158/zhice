```java
     * 保存测试结果

        result.setTypePScore(personalityResults.getTypePScore());
        result.setTypeMScore(personalityResults.getTypeMScore());
        result.setTypeSScore(personalityResults.getTypeSScore());

        personalityResultService.save(result);
        
        return R.ok(result.getResultId(), "保存成功");

     * 获取用户最近一次测试结果

     * @param userId 用户ID
```

```java
/**
 * 人格测试记录控制器
 * 
 * @author ruoyi

@RestController
@RequestMapping("/personality/records")
public class PersonalityRecordController extends BaseController {
    
    @Autowired
    private IPersonalityRecordService personalityRecordService;
    
    @Autowired
    private IPersonalityAnswerService personalityAnswerService;
    
    /**
     * 提交人格测试
     * 
     * @param personalityTestRequest 测试请求对象
     * @return 操作结果
     */
    @PostMapping("/add")
    public R<?> submitPersonalityTest(@RequestBody PersonalityTestRequest request) {
        Long userId = request.getUserId();
        List<Integer> selectedOptionIds = request.getSelectedOptionIds();
        
        // 1. 插入记录
        Long recordId = personalityRecordService.saveRecord(userId);
        
        // 2. 构建答案列表
        List<PersonalityAnswer> answers = new ArrayList<>();
        for (int i = 0; i < selectedOptionIds.size(); i++) {
            PersonalityAnswer answer = new PersonalityAnswer();
            answer.setRecordId(recordId);
            answer.setOptionId(Long.valueOf(selectedOptionIds.get(i)));
            answers.add(answer);
        }
        
        // 3. 批量插入答案
        personalityAnswerService.saveAll(answers);
        
        return R.ok(recordId, "提交成功");
    }
}
```

#### 2.1.4 算法设计

**评分算法：**

```java
/**
 * 计算人格评分
 * 
 * @param recordId 记录ID
 * @return 评分结果
 */
public PersonalityScoreResponse calculateScores(Integer recordId) {
    // 1. 查询用户的答案
    List<PersonalityAnswer> answers = personalityAnswerService.getByRecordId(
        Long.valueOf(recordId)
    );
    
    // 2. 查询每个选项对应的维度和分值
    Map<Long, Integer> optionScores = new HashMap<>();
    for (PersonalityAnswer answer : answers) {
        PersonalityOptions option = personalityOptionsService.getById(
            answer.getOptionId()
        );
        optionScores.put(answer.getOptionId(), option.getScore());
    }
    
    // 3. 按维度汇总分数
    int typeCScore = 0;
    int typePScore = 0;
    int typeMScore = 0;
    int typeSScore = 0;
    
    for (Map.Entry<Long, Integer> entry : optionScores.entrySet()) {
        // 根据选项ID的规则判断维度
        int optionId = entry.getKey();
        int score = entry.getValue();
        
        if (optionId % 4 == 1) {
            typeCScore += score;
        } else if (optionId % 4 == 2) {
            typePScore += score;
        } else if (optionId % 4 == 3) {
            typeMScore += score;
        } else {
            typeSScore += score;
        }
    }
    
    // 4. 查询映射表确定人格类型
    PersonalityTypes type = determinePersonalityType(
        typeCScore, typePScore, typeMScore, typeSScore
    );
    
    // 5. 构建响应对象
    PersonalityScoreResponse response = new PersonalityScoreResponse();
    response.setTypeCScore(typeCScore);
    response.setTypePScore(typePScore);
    response.setTypeMScore(typeMScore);
    response.setTypeSScore(typeSScore);
    response.setTypeCode(type.getTypeCode());
    response.setTypeName(type.getTypeName());
    response.setDescription(type.getDescription());
    
    return response;
}

/**
 * 确定人格类型
 * 
 * @param cScore C得分
 * @param pScore P得分
 * @param mScore M得分
 * @param sScore S得分
 * @return 人格类型
 */
private PersonalityTypes determinePersonalityType(int cScore, int pScore, 
                                                   int mScore, int sScore) {
    // 根据最高分确定主导类型
    int maxScore = Math.max(Math.max(cScore, pScore), Math.max(mScore, sScore));
    
    if (maxScore == cScore) {
        return personalityTypesService.getByTypeCode("C");
    } else if (maxScore == pScore) {
        return personalityTypesService.getByTypeCode("P");
    } else if (maxScore == mScore) {
        return personalityTypesService.getByTypeCode("M");
    } else {
        return personalityTypesService.getByTypeCode("S");
    }
}
```

```java
public Integer processAndStoreAnswers(List<AnswerDTO.AnswerItem> answers, Long userId) {
    // 1. 保存测试记录
    PersonalityRecord record = new PersonalityRecord();
    record.setUserId(userId);
    record.setTestDate(new Date());
    record.setStatus(1); // 已完成
    personalityRecordMapper.insert(record);
    
    // 2. 保存答案
    List<PersonalityAnswer> answerList = new ArrayList<>();
    for (AnswerDTO.AnswerItem item : answers) {
        PersonalityAnswer answer = new PersonalityAnswer();
        answer.setRecordId(record.getRecordId());
        answer.setOptionId(item.getOptionId());
        answerList.add(answer);
    }
    personalityAnswerMapper.batchInsert(answerList);
    
    // 3. 返回记录ID
    return record.getRecordId().intValue();
}
```

```java
/**
 * 全局异常处理器
 * 
 * @author ruoyi
 */
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {
    
    /**
     * 参数验证异常
     */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public AjaxResult handleMethodArgumentNotValidException(
            MethodArgumentNotValidException e) {
        log.error("参数验证异常", e);
        String message = e.getBindingResult().getFieldError().getDefaultMessage();
        return AjaxResult.error(message);
    }
    
    /**
     * 业务异常
     */
    @ExceptionHandler(ServiceException.class)
    public AjaxResult handleServiceException(ServiceException e) {
        log.error("业务异常", e);
        Integer code = e.getCode();
        return StringUtils.isNotNull(code) ? AjaxResult.error(code, e.getMessage())
            : AjaxResult.error(e.getMessage());
    }
    
    /**
     * 认证异常
     */
    @ExceptionHandler(AuthenticationException.class)
    public AjaxResult handleAuthenticationException(AuthenticationException e) {
        log.error("认证异常", e);
        return AjaxResult.error(401, "认证失败");
    }
    
    /**
     * 系统异常
     */
    @ExceptionHandler(Exception.class)
    public AjaxResult handleException(Exception e) {
        log.error("系统异常", e);
        return AjaxResult.error("系统异常，请联系管理员");
    }
}
```

### 6.2 XSS过滤

```java
/**
 * XSS过滤器
 * 
 * @author ruoyi
 */
@Component
public class XssFilter implements Filter {
    
    @Override
    public void doFilter(ServletRequest request, ServletResponse response, 
                        FilterChain chain) throws IOException, ServletException {
        XssHttpServletRequestWrapper xssRequest = 
            new XssHttpServletRequestWrapper((HttpServletRequest) request);
        chain.doFilter(xssRequest, response);
    }
}

/**
 * XSS请求包装类
 */
public class XssHttpServletRequestWrapper extends HttpServletRequestWrapper {
    
    @Override
    public String getParameter(String name) {
        String value = super.getParameter(name);
        return cleanXSS(value);
    }
    
    @Override
    public String[] getParameterValues(String name) {
        String[] values = super.getParameterValues(name);
        if (values == null) {
            return null;
        }
        int count = values.length;
        String[] encodedValues = new String[count];
        for (int i = 0; i < count; i++) {
            encodedValues[i] = cleanXSS(values[i]);
        }
        return encodedValues;
    }
    
    /**
     * 清理XSS
     */
    private String cleanXSS(String value) {
        if (value == null) {
            return null;
        }
        value = value.replaceAll("<", "&lt;").replaceAll(">", "&gt;");
        value = value.replaceAll("\\(", "&#40;").replaceAll("\\)", "&#41;");
        value = value.replaceAll("'", "&#39;");
        value = value.replaceAll("eval\\((.*)\\)", "");
        value = value.replaceAll("[\\\"\\\'][\\s]*javascript:(.*)[\\\"\\\']", "\"\"");
        value = value.replaceAll("script", "");
        return value;
    }
}
```
